WKLEJ TO W FIREBASE → Firestore Database → Rules → Publish

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }
    function isAdmin() {
      return signedIn() && request.auth.token.email == "contact@mkulczyk.pl";
    }
    function isOwnerEmail(email) {
      return signedIn() && email == request.auth.token.email;
    }

    match /admin/{docId} {
      allow read, write: if isAdmin();
    }

    match /settings/{docId} {
      allow read, write: if isAdmin();
    }

    match /clients/{clientId} {
      allow read: if isAdmin() || isOwnerEmail(resource.data.email);
      allow create: if isAdmin();
      allow update: if isAdmin() || (
        isOwnerEmail(resource.data.email) &&
        request.resource.data.email == resource.data.email
      );
      allow delete: if isAdmin();
    }

    match /requests/{requestId} {
      allow read: if isAdmin() || isOwnerEmail(resource.data.clientEmail);
      allow create: if isAdmin() || isOwnerEmail(request.resource.data.clientEmail);
      allow update: if isAdmin() || (
        isOwnerEmail(resource.data.clientEmail) &&
        request.resource.data.clientEmail == resource.data.clientEmail
      );
      allow delete: if isAdmin();
    }

    match /notifications/{notificationId} {
      allow read, update, delete: if isAdmin();
      allow create: if signedIn()
        && request.resource.data.clientEmail == request.auth.token.email
        && request.resource.data.targetEmail == "contact@mkulczyk.pl";
    }
  }
}
